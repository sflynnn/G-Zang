#### 1.2.2 包命名规则
- **包名全小写**：使用小写字母和数字
- **层次分明**：按功能模块划分子包
- **避免冲突**：使用公司域名倒置作为根包名
- **语义明确**：包名应能反映其功能职责

```java
// ✅ 正确示例
com.gzang.app.controller
com.gzang.app.service.impl
com.gzang.app.mapper.user

// ❌ 错误示例
com.gzang.App.Controller    // 大写字母
com.gzang.app.ctl           // 缩写不清晰
```

### 2.2 异常处理规范

#### 2.2.1 统一异常处理
```java
@RestControllerAdvice
@Slf4j
public class GlobalExceptionHandler {

    /**
     * 处理业务异常
     */
    @ExceptionHandler(BusinessException.class)
    public Result<Void> handleBusinessException(BusinessException e) {
        log.warn("业务异常: {}", e.getMessage());
        return Result.error(e.getCode(), e.getMessage());
    }

    /**
     * 处理参数验证异常
     */
    @ExceptionHandler(MethodArgumentNotValidException.class)
    public Result<Void> handleValidationException(MethodArgumentNotValidException e) {
        String message = e.getBindingResult().getFieldErrors().stream()
            .map(error -> error.getField() + ": " + error.getDefaultMessage())
            .collect(Collectors.joining(", "));
        log.warn("参数验证失败: {}", message);
        return Result.error(400, "参数错误: " + message);
    }

    /**
     * 处理系统异常
     */
    @ExceptionHandler(Exception.class)
    public Result<Void> handleSystemException(Exception e) {
        log.error("系统异常", e);
        return Result.error(500, "系统内部错误");
    }
}
```

```java
public interface UserMapper extends BaseMapper<User> {

     */
    int updateStatusBatch(@Param("ids") List<Long> ids, @Param("status") Integer status);

    List<Map<String, Object>> countByStatus();
```

#### 3.1.2 XML 映射文件
```xml
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="com.gzang.app.mapper.UserMapper">

    <!-- 根据用户名查询用户 -->
    <select id="selectByUsername" resultType="com.gzang.app.entity.User">
        SELECT * FROM t_user WHERE username = #{username}
    </select>

    <!-- 分页查询用户 -->
    <select id="selectUserPage" resultType="com.gzang.app.entity.User">
        SELECT * FROM t_user
        <where>
            <if test="query.username != null and query.username != ''">
                AND username LIKE CONCAT('%', #{query.username}, '%')
            </if>
            <if test="query.status != null">
                AND status = #{query.status}
            </if>
            <if test="query.roleId != null">
                AND role_id = #{query.roleId}
            </if>
        </where>
        ORDER BY create_time DESC
    </select>

    <!-- 批量更新状态 -->
    <update id="updateStatusBatch">
        UPDATE t_user SET status = #{status}, update_time = NOW()
        WHERE id IN
        <foreach item="id" collection="ids" open="(" separator="," close=")">
            #{id}
        </foreach>
    </update>

    <!-- 统计用户状态 -->
    <select id="countByStatus" resultType="map">
        SELECT status, COUNT(*) as count FROM t_user GROUP BY status
    </select>

</mapper>
```

#### 3.1.3 条件构造器使用
```java
@Service
public class UserServiceImpl implements UserService {

    @Autowired
    private UserMapper userMapper;

    public IPage<User> getUserPage(UserQuery query, int page, int size) {
        Page<User> pageParam = new Page<>(page, size);

        return userMapper.selectPage(pageParam, new LambdaQueryWrapper<User>()
            .like(StringUtils.hasText(query.getUsername()), User::getUsername, query.getUsername())
            .eq(query.getStatus() != null, User::getStatus, query.getStatus())
            .eq(query.getRoleId() != null, User::getRoleId, query.getRoleId())
            .orderByDesc(User::getCreateTime)
        );
    }

    public boolean updateUserStatus(Long userId, Integer status) {
        return userMapper.update(null, new LambdaUpdateWrapper<User>()
            .set(User::getStatus, status)
            .set(User::getUpdateTime, LocalDateTime.now())
            .eq(User::getId, userId)
        ) > 0;
    }
}
```

#### 4.1.2 实现类规范
```java
/**
 * 用户服务实现类
 */
@Slf4j
@Service
@Transactional(readOnly = true)
public class UserServiceImpl implements UserService {

    @Autowired
    private UserMapper userMapper;

    @Autowired
    private RoleMapper roleMapper;

    @Autowired
    private CompanyMapper companyMapper;

    @Autowired
    private PasswordEncoder passwordEncoder;

    @Override
    public UserVO getUserById(Long id) {
        log.debug("获取用户详情，ID: {}", id);

        User user = userMapper.selectById(id);
        if (user == null) {
            throw new DataNotFoundException("用户");
        }

        // 获取关联数据
        Role role = roleMapper.selectById(user.getRoleId());
        Company company = null;
        if (user.getCompanyId() != null) {
            company = companyMapper.selectById(user.getCompanyId());
        }

        return UserVO.fromEntityWithRelations(user, role, company);
    }

    @Override
    public User getUserByUsername(String username) {
        return userMapper.selectByUsername(username);
    }

    @Override
    public IPage<UserVO> getUserPage(UserQuery query, int page, int size) {
        Page<User> pageParam = new Page<>(page, size);
        IPage<User> userPage = userMapper.selectUserPage(pageParam, query);

        // 转换为VO
        return userPage.convert(user -> {
            Role role = roleMapper.selectById(user.getRoleId());
            Company company = user.getCompanyId() != null ?
                companyMapper.selectById(user.getCompanyId()) : null;
            return UserVO.fromEntityWithRelations(user, role, company);
        });
    }

    @Override
    @Transactional
    public UserVO createUser(CreateUserDTO dto) {
        log.info("创建用户: {}", dto.getUsername());

        // 检查用户名是否已存在
        if (userMapper.selectByUsername(dto.getUsername()) != null) {
            throw new BusinessException(400, "用户名已存在");
        }

        // 创建用户实体
        User user = User.builder()
            .username(dto.getUsername())
            .password(passwordEncoder.encode(dto.getPassword()))
            .nickname(dto.getNickname())
            .email(dto.getEmail())
            .roleId(3) // 默认普通用户角色
            .status(1) // 默认正常状态
            .build();

        userMapper.insert(user);
        log.info("用户创建成功，ID: {}", user.getId());

        return UserVO.fromEntity(user);
    }

    @Override
    @Transactional
    public UserVO updateUser(Long id, UpdateUserDTO dto) {
        log.info("更新用户，ID: {}", id);

        User user = userMapper.selectById(id);
        if (user == null) {
            throw new DataNotFoundException("用户");
        }

        // 更新字段
        if (StringUtils.hasText(dto.getNickname())) {
            user.setNickname(dto.getNickname());
        }
        if (StringUtils.hasText(dto.getAvatar())) {
            user.setAvatar(dto.getAvatar());
        }

        userMapper.updateById(user);

        return UserVO.fromEntity(user);
    }

    @Override
    @Transactional
    public void deleteUser(Long id) {
        log.info("删除用户，ID: {}", id);

        if (userMapper.selectById(id) == null) {
            throw new DataNotFoundException("用户");
        }

        userMapper.deleteById(id);
    }

    @Override
    @Transactional
    public boolean updateUserStatus(Long id, Integer status) {
        log.info("更新用户状态，ID: {}, 状态: {}", id, status);

        return userMapper.updateStatusBatch(Arrays.asList(id), status) > 0;
    }
}
```

```java
@Service
public class AccountingServiceImpl implements AccountingService {

    @Autowired
    private TransactionMapper transactionMapper;

    @Autowired
    private AccountMapper accountMapper;

    /**
     * 创建交易（需要事务保证数据一致性）
     */
    @Override
    @Transactional
    public TransactionVO createTransaction(CreateTransactionDTO dto) {
        // 1. 验证账户存在
        Account account = accountMapper.selectById(dto.getAccountId());
        if (account == null) {
            throw new DataNotFoundException("账户");
        }

        // 2. 验证分类存在
        Category category = categoryMapper.selectById(dto.getCategoryId());
        if (category == null) {
            throw new DataNotFoundException("分类");
        }

        // 3. 创建交易记录
        Transaction transaction = Transaction.builder()
            .amount(dto.getAmount())
            .type(dto.getType())
            .categoryId(dto.getCategoryId())
            .accountId(dto.getAccountId())
            .transactionTime(dto.getTransactionTime())
            .remark(dto.getRemark())
            .build();

        transactionMapper.insert(transaction);

        // 4. 更新账户余额
        BigDecimal newBalance = calculateNewBalance(account.getBalance(), dto);
        accountMapper.updateBalance(dto.getAccountId(), newBalance);

        return TransactionVO.fromEntity(transaction);
    }

    /**
     * 只读操作使用 readOnly 事务
     */
    @Override
    @Transactional(readOnly = true)
    public IPage<TransactionVO> getTransactionPage(TransactionQuery query, int page, int size) {
        Page<Transaction> pageParam = new Page<>(page, size);
        return transactionMapper.selectTransactionPage(pageParam, query)
            .convert(TransactionVO::fromEntity);
    }

    /**
     * 计算新余额
     */
    private BigDecimal calculateNewBalance(BigDecimal currentBalance, CreateTransactionDTO dto) {
        if (dto.getType() == 1) { // 收入
            return currentBalance.add(dto.getAmount());
        } else { // 支出
            return currentBalance.subtract(dto.getAmount());
        }
    }
}
```

```java
            .message("操作成功")
            .data(data)
            .timestamp(System.currentTimeMillis())
            .build();
    }

    /**
     * 成功响应（无数据）
     */
    public static <T> Result<T> success() {
        return success(null);
    }

    /**
     * 错误响应

    public static <T> Result<T> error(String message) {
        return error(500, message);
    }

    /**
     * 错误响应（指定状态码）

    public static <T> Result<T> error(Integer code, String message) {
```
