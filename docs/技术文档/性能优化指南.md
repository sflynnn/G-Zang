```typescript
// composables/useDebounce.ts
import { ref, Ref } from 'vue'

export function useDebounce<T extends (...args: any[]) => any>(
  func: T,
  delay: number
): T {
  let timeoutId: NodeJS.Timeout

  return ((...args: Parameters<T>) => {
    clearTimeout(timeoutId)
    timeoutId = setTimeout(() => func(...args), delay)
  }) as T
}

// composables/useThrottle.ts
export function useThrottle<T extends (...args: any[]) => any>(
  func: T,
  limit: number
): T {
  let inThrottle: boolean

  return ((...args: Parameters<T>) => {
    if (!inThrottle) {
      func(...args)
      inThrottle = true
      setTimeout(() => inThrottle = false, limit)
    }
  }) as T
}
```

```typescript
// composables/useTimeout.ts
export function useTimeout(callback: () => void, delay: number) {
  const timeoutId = ref<NodeJS.Timeout>()

  const start = () => {
    timeoutId.value = setTimeout(callback, delay)
  }

  const stop = () => {
    if (timeoutId.value) {
      clearTimeout(timeoutId.value)
      timeoutId.value = undefined
    }
  }

  onUnmounted(() => {
    stop()
  })

  return { start, stop }
}

export function useIntersectionObserver(
  target: Ref<Element | null>,
  callback: IntersectionObserverCallback,
  options?: IntersectionObserverInit
) {
  let observer: IntersectionObserver | null = null

  onMounted(() => {
    if (target.value) {
      observer = new IntersectionObserver(callback, options)
      observer.observe(target.value)
    }
  })

  onUnmounted(() => {
    if (observer) {
      observer.disconnect()
      observer = null
    }
  })
```

#### 2.3.2 应用级缓存
```typescript
// stores/useCache.ts
import { ref, computed } from 'vue'

interface CacheItem<T> {
  data: T
  timestamp: number
  ttl: number // Time To Live in milliseconds
}

class MemoryCache {
  private cache = new Map<string, CacheItem<any>>()

  set<T>(key: string, data: T, ttl = 5 * 60 * 1000): void { // 默认5分钟
    this.cache.set(key, {
      data,
      timestamp: Date.now(),
      ttl
    })
  }

  get<T>(key: string): T | null {
    const item = this.cache.get(key)
    if (!item) return null

    if (Date.now() - item.timestamp > item.ttl) {
      this.cache.delete(key)
      return null
    }

    return item.data
  }

  clear(): void {
    this.cache.clear()
  }

  delete(key: string): void {
    this.cache.delete(key)
  }
}

export const memoryCache = new MemoryCache()

// 使用示例
const loadCategories = async () => {
  const cacheKey = 'categories'
  let categories = memoryCache.get<Category[]>(cacheKey)

  if (!categories) {
    const response = await api.getCategories()
    categories = response.data
    memoryCache.set(cacheKey, categories, 10 * 60 * 1000) // 10分钟缓存
  }

  return categories
}
```

```java
public class GcTuningConfig {

    @PostConstruct
    public void initGcMonitoring() {
        // 注册GC监听器
        List<GarbageCollectorMXBean> gcBeans = ManagementFactory.getGarbageCollectorMXBeans();

        for (GarbageCollectorMXBean gcBean : gcBeans) {
            NotificationEmitter emitter = (NotificationEmitter) gcBean;
            emitter.addNotificationListener((notification, handback) -> {
                if (notification.getType().equals("com.sun.management.gc.notification")) {
                    // 记录GC事件
                    logGcEvent(notification);
                }
            }, null, null);
        }
    }

        CompositeData cd = (CompositeData) notification.getUserData();

        // 提取GC信息
        String gcName = (String) cd.get("gcName");
        long duration = (Long) cd.get("gcInfo.duration");
        long startTime = (Long) cd.get("gcInfo.startTime");
```

```java
@Service
public class CachedCategoryService {

    @Autowired
    private CategoryMapper categoryMapper;

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    /**
     * 缓存穿透防护
     */
    @Cacheable(value = "categories", key = "#userId + '_' + #companyId",
               unless = "#result.isEmpty()")
    public List<Category> getCategories(Long userId, Long companyId) {
        List<Category> categories = categoryMapper.selectByUserOrCompany(userId, companyId);

        // 如果查询结果为空，也缓存一个空列表，避免缓存穿透
        if (categories.isEmpty()) {
            categories = Collections.emptyList();
        }

        return categories;
    }

    /**
     * 缓存雪崩防护 - 随机过期时间
     */
    @Cacheable(value = "categoryTree", key = "#userId + '_' + #companyId")
    public CategoryTree getCategoryTree(Long userId, Long companyId) {
        CategoryTree tree = buildCategoryTree(userId, companyId);

        // 设置随机过期时间，避免缓存雪崩
        String cacheKey = "gzang:categoryTree:" + userId + "_" + companyId;
        int randomExpire = 1800 + new Random().nextInt(600); // 30-40分钟随机过期
        redisTemplate.expire(cacheKey, randomExpire, TimeUnit.SECONDS);

        return tree;
    }

    /**
     * 双写一致性保证
     */
    @Transactional
    public void updateCategory(Category category) {
        // 先更新数据库
        categoryMapper.updateById(category);

        // 再删除缓存，保证数据一致性
        String pattern = "gzang:categories:*";
        Set<String> keys = redisTemplate.keys(pattern);
        if (!keys.isEmpty()) {
            redisTemplate.delete(keys);
        }

        // 也可以使用@CacheEvict注解
        // @CacheEvict(value = "categories", allEntries = true)
    }
```

```java
    public DataSource masterDataSource() {

    }

    @Bean
    @ConfigurationProperties(prefix = "spring.datasource.slave")

        return DataSourceBuilder.create().build();
    }

    @Bean
    public DataSource routingDataSource() {
        Map<Object, Object> targetDataSources = new HashMap<>();
        targetDataSources.put(DataSourceType.MASTER, masterDataSource());
        targetDataSources.put(DataSourceType.SLAVE, slaveDataSource());

        RoutingDataSource routingDataSource = new RoutingDataSource();
        routingDataSource.setDefaultTargetDataSource(masterDataSource());
        routingDataSource.setTargetDataSources(targetDataSources);

        return routingDataSource;
    }

    @Bean
    public DataSource dataSource() {
        return new LazyConnectionDataSourceProxy(routingDataSource());
    }
}

public class RoutingDataSource extends AbstractRoutingDataSource {

    @Override
    protected Object determineCurrentLookupKey() {
        return DataSourceContextHolder.getDataSourceType();
    }
}

public class DataSourceContextHolder {
    private static final ThreadLocal<DataSourceType> CONTEXT = new ThreadLocal<>();

    public static void setDataSourceType(DataSourceType type) {
        CONTEXT.set(type);
    }

    public static DataSourceType getDataSourceType() {
        return CONTEXT.get() != null ? CONTEXT.get() : DataSourceType.MASTER;
    }

    public static void clear() {
        CONTEXT.remove();
    }
}

public enum DataSourceType {
    MASTER, SLAVE
}
```

```java
@Configuration
public class ShardingConfig {

    @Bean
    public DataSource shardingDataSource() {
        // 配置分库分表规则
        ShardingRuleConfiguration shardingRuleConfig = new ShardingRuleConfiguration();

        // 交易表分库分表规则
        ShardingTableRuleConfiguration transactionRule = new ShardingTableRuleConfiguration(
            "t_transaction", "ds_${0..3}.t_transaction_${0..15}");

        // 行表达式分片策略
        transactionRule.setTableShardingStrategy(new StandardShardingStrategyConfiguration(
            "user_id", new TransactionTableShardingAlgorithm()));

        // 数据库分片策略
        transactionRule.setDatabaseShardingStrategy(new StandardShardingStrategyConfiguration(
            "user_id", new DatabaseShardingAlgorithm()));

        shardingRuleConfig.getTables().add(transactionRule);

        // 创建数据源
        return ShardingDataSourceFactory.createDataSource(
            createDataSourceMap(), shardingRuleConfig, new Properties());
    }

    private Map<String, DataSource> createDataSourceMap() {
        Map<String, DataSource> dataSourceMap = new HashMap<>();
        for (int i = 0; i < 4; i++) {
            dataSourceMap.put("ds_" + i, createDataSource("db_" + i));
        }
        return dataSourceMap;
    }
}

public class TransactionTableShardingAlgorithm implements StandardShardingAlgorithm<Long> {

    @Override
    public String doSharding(Collection<String> availableTargetNames, PreciseShardingValue<Long> shardingValue) {
        Long userId = shardingValue.getValue();
        // 根据用户ID分16个表
        int tableIndex = (int) (userId % 16);
        return "t_transaction_" + tableIndex;
    }

    @Override
    public Collection<String> doSharding(Collection<String> availableTargetNames, RangeShardingValue<Long> shardingValue) {
        // 范围查询需要返回所有可能的表
        return availableTargetNames;
    }
}
```
